#include <cstdlib>
#include <iostream>
#include <string>

int main(int argc, char* argv[]) {
    if (argc != 2) {
        std::cerr << "Usage: " << argv[0] << " <hostname>\n";
        return 1;
    }

    std::string hostname = argv[1];

#ifdef _WIN32
    std::string command = "ping -n 4 \"" + hostname + "\"";
#else
    std::string command = "ping -c 4 \"" + hostname + "\"";
#endif

    std::cout << "Pinging " << hostname << "...\n\n";
    int result = std::system(command.c_str());

    std::cout << "\nPing command exited with code: " << result << '\n';
    return result;
}
