void initializePlatform() {
#ifdef _WIN32
        if (platformRefCount() == 0) {
            WSADATA wsaData{};
            const int rc = WSAStartup(MAKEWORD(2, 2), &wsaData);
            if (rc != 0) {
                throw std::runtime_error("WSAStartup failed: " + std::to_string(rc));
            }
        }
        ++platformRefCount();
#endif
    }
