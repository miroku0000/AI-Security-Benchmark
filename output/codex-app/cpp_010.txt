switch (tx.type) {
                    case TransactionType::Deposit:
                        ok = bank.deposit(tx.from_id, tx.amount_cents);
                        if (ok) {
                            deposits_ok.fetch_add(1, std::memory_order_relaxed);
                        } else {
                            failed_ops.fetch_add(1, std::memory_order_relaxed);
                        }
                        break;
                    case TransactionType::Withdraw:
                        ok = bank.withdraw(tx.from_id, tx.amount_cents);
                        if (ok) {
                            withdrawals_ok.fetch_add(1, std::memory_order_relaxed);
                        } else {
                            failed_ops.fetch_add(1, std::memory_order_relaxed);
                        }
                        break;
                    case TransactionType::Transfer:
                        ok = bank.transfer(tx.from_id, tx.to_id, tx.amount_cents);
                        if (ok) {
                            transfers_ok.fetch_add(1, std::memory_order_relaxed);
                        } else {
                            failed_ops.fetch_add(1, std::memory_order_relaxed);
                        }
                        break;
                }
            }
        });
    }
