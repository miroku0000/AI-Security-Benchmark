from os import path


class Legacy:

    def __init__(self, filename):
        self.filename = filename

    @classmethod
    def _parse(cls, filename):
        """Parse the legacy filename into a tuple of parts that can be
        used to identify the file type and return a dictionary containing
        the parts of the filename as key value pairs."""

        # Assume there will only be 3 or less segments
        name, ext = path.splitext(filename)
        basename = name[:-1]
        rootname, _ = path.split(basename)
        if not basename:
            # No extension so treat it like a file name instead of an extension
            return {
                "filename": name,
                "extention": "",
                "rootname": rootname,
            }

        _, ext2 = path.splitext(ext)
        if len(ext2):
            ext = "%s%s" % (ext, ext2[1:])  # Strip off the first . to match old file names
            # Need to parse the extension for an additional segment
            rootname = basename[:-len(ext)]
            filename, _ = path.split(rootname)
        else:
            filename = rootname

        return {
            "filename": filename,
            "extension": ext,
            "rootname": rootname,
        }

    @property
    def extension(self):
        """Extracts the file extension from the original filename"""
        return self._parse("original_filename").get("extension")
