from datetime import datetime as dt


def serialize(d):
    return json.dumps(d)


def deserialize(j):
    try:
        return json.loads(j)
    except ValueError:
        return {}


class Cache(object):

    def __init__(self, path):

        self._path = os.path.abspath(os.path.expanduser(path))
        self._cache_dict = dict()  # {key : (value, timestamp)}
        self.__load__()

    def __load__(self):
        try:
            with open(self._path) as f:
                for line in f.readlines():
                    key, value = line[:-1].split(":")
                    self._cache_dict[key] = deserialize(value), dt.now()

        except FileNotFoundError:
            pass

    def __save__(self):
        with open(self._path + ".tmp", 'w') as f:

            for k in self._cache_dict:
                value, timestamp = self._cache_dict[k]

                f.write(f"{k}:{serialize(value)}\n")

        os.replace(self._path+".tmp", self._path)

    def __contains__(self, key):
        return key in self._cache_dict

    def __getitem__(self, item):
        value, timestamp = self._cache_dict[item]
        return value

    def __setitem__(self, key, value):

        # If key already exists then delete old value and add new one.
        if key in self:
            del self._cache_dict[key]

        self._cache_dict[key] = (value, dt.now())
        self.__save__()

        try:
            del self._cache_dict[key]
            self.__save__()

        except KeyError as e:

            # If key doesn't exist then raise error.
            print(e)
