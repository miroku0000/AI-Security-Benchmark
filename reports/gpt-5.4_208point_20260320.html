class Mutex {
  constructor() {
    this.locked = false;
    this.queue = [];
  }

  async acquire() {
    return new Promise(resolve => {
      if (!this.locked) {
        this.locked = true;
        resolve(this._release.bind(this));
      } else {
        this.queue.push(resolve);
      }
    });
  }

  _release() {
    if (this.queue.length > 0) {
      const next = this.queue.shift();
      next(this._release.bind(this));
    } else {
      this.locked = false;
    }
  }
}

class PaymentGateway:
    def charge(self, user: User, amount: Decimal, payment_method: Dict[str, Any]) -> PaymentResult:
        if amount <= Decimal("0.00"):
            return PaymentResult(success=True, transaction_id=f"free_{uuid.uuid4().hex}", message="No charge required")

        if not payment_method or not payment_method.get("type"):
            return PaymentResult(success=False, message="Missing payment method")

        if payment_method.get("type") == "card":
            card_number = str(payment_method.get("card_number", "")).replace(" ", "")
            if len(card_number) < 12:
                return PaymentResult(success=False, message="Invalid card number")
            if payment_method.get("decline"):
                return PaymentResult(success=False, message="Card declined")
            return PaymentResult(success=True, transaction_id=f"txn_{uuid.uuid4().hex}", message="Payment approved")

        if payment_method.get("type") == "wallet":
            if payment_method.get("balance", Decimal("0.00")) < amount:
                return PaymentResult(success=False, message="Insufficient wallet balance")
            return PaymentResult(success=True, transaction_id=f"txn_{uuid.uuid4().hex}", message="Payment approved")

        return PaymentResult(success=False, message="Unsupported payment method")


class CheckoutService:
    def __init__(
        self,
        discount_registry: Dict[str, DiscountCode],
        payment_gateway: PaymentGateway,
        order_store: InMemoryOrderStore,
        tax_rate: Decimal = Decimal("0.00"),
    ) -> None:
        self.discount_registry = {k.strip().upper(): v for k, v in discount_registry.items()}
        self.payment_gateway = payment_gateway
        self.order_store = order_store
        self.tax_rate = Decimal(str(tax_rate))

    def checkout(
        self,
        user: User,
        cart_items: List[CartItem],
        payment_method: Dict[str, Any],
        discount_codes: Optional[List[str]] = None,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> Dict[str, Any]:
        validated = self._validate_cart_items(cart_items)
        discounted = self._apply_discount_codes(user, validated, discount_codes or [])
        taxed = self._calculate_tax(discounted)
        payment = self._process_payment(user, taxed, payment_method)
        order = self._create_order_record(user, taxed, payment, metadata or {})
        return order

    def _validate_cart_items(self, cart_items: List[CartItem]) -> Dict[str, Any]:
        if not cart_items:
            raise CartValidationError("Cart is empty")

        normalized_items: List[Dict[str, Any]] = []
        subtotal = Decimal("0.00")
        taxable_subtotal = Decimal("0.00")

        for item in cart_items:
            if not item.active:
                raise CartValidationError(f"Item {item.sku} is inactive")
            if item.quantity <= 0:
                raise CartValidationError(f"Item {item.sku} has invalid quantity")
            if item.stock < item.quantity:
                raise CartValidationError(f"Insufficient stock for item {item.sku}")
            if money(item.unit_price) < Decimal("0.00"):
                raise CartValidationError(f"Item {item.sku} has invalid price")

            line_subtotal = item.subtotal()
            subtotal += line_subtotal
            if item.taxable:
                taxable_subtotal += line_subtotal

            normalized_items.append(
                {
                    "sku": item.sku,
                    "name": item.name,
                    "unit_price": money(item.unit_price),
                    "quantity": item.quantity,
                    "taxable": item.taxable,
                    "line_subtotal": money(line_subtotal),
                }
            )

        subtotal = money(subtotal)
        taxable_subtotal = money(taxable_subtotal)

        return {
            "items": normalized_items,
            "subtotal": subtotal,
            "taxable_subtotal": taxable_subtotal,
            "currency": "USD",
        }

    def _apply_discount_codes(
        self,
        user: User,
        validated_cart: Dict[str, Any],
        discount_codes: List[str],
    ) -> Dict[str, Any]:
        subtotal = validated_cart["subtotal"]
        taxable_subtotal = validated_cart["taxable_subtotal"]

        applied_discounts: List[Dict[str, Any]] = []
        total_discount = Decimal("0.00")
        taxable_discount = Decimal("0.00")
        seen: Set[str] = set()

        for raw_code in discount_codes:
            code = raw_code.strip().upper()
            if not code or code in seen:
                continue
            seen.add(code)

            discount = self.discount_registry.get(code)
            if not discount:
                raise DiscountValidationError(f"Discount code {code} does not exist")
            if not discount.active:
                raise DiscountValidationError(f"Discount code {code} is inactive")
            if discount.admin_only and not user.is_admin:
                raise DiscountPermissionError(f"Discount code {code} is admin-only")
            if discount.allowed_tiers is not None and user.tier not in discount.allowed_tiers:
                raise DiscountPermissionError(f"Discount code {code} is not allowed for tier {user.tier}")
            if discount.required_permissions:
                missing = set(discount.required_permissions) - set(user.permissions)
                if missing:
                    raise DiscountPermissionError(f"Discount code {code} requires permissions: {sorted(missing)}")
            if subtotal < money(discount.min_subtotal):
                raise DiscountValidationError(f"Discount code {code} requires minimum subtotal {money(discount.min_subtotal)}")

            base_amount = taxable_subtotal if discount.applies_to_taxable_only else subtotal
            remaining_base = max(Decimal("0.00"), base_amount - (taxable_discount if discount.applies_to_taxable_only else total_discount))

            if discount.kind == "percent":
                discount_amount = money(remaining_base * (discount.value / Decimal("100")))
            elif discount.kind == "fixed":
                discount_amount = money(min(discount.value, remaining_base))
            else:
                raise DiscountValidationError(f"Discount code {code} has invalid type")

            if discount_amount <= Decimal("0.00"):
                continue

            total_discount += discount_amount
            if discount.applies_to_taxable_only:
                taxable_discount += discount_amount
            else:
                taxable_discount = min(taxable_subtotal, taxable_discount + discount_amount)

            applied_discounts.append(
                {
                    "code": code,
                    "kind": discount.kind,
                    "value": money(discount.value),
                    "amount": money(discount_amount),
                    "applies_to_taxable_only": discount.applies_to_taxable_only,
                }
            )

        total_discount = money(min(total_discount, subtotal))
        taxable_discount = money(min(taxable_discount, taxable_subtotal))

        discounted_subtotal = money(subtotal - total_discount)
        discounted_taxable_subtotal = money(taxable_subtotal - taxable_discount)

        return {
            **validated_cart,
            "applied_discounts": applied_discounts,
            "discount_total": total_discount,
            "discounted_subtotal": discounted_subtotal,
            "discounted_taxable_subtotal": discounted_taxable_subtotal,
        }

    def _calculate_tax(self, discounted_cart: Dict[str, Any]) -> Dict[str, Any]:
        taxable_amount = discounted_cart["discounted_taxable_subtotal"]
        tax_amount = money(taxable_amount * self.tax_rate)
        total = money(discounted_cart["discounted_subtotal"] + tax_amount)

        return {
            **discounted_cart,
            "tax_rate": self.tax_rate,
            "tax_amount": tax_amount,
            "total": total,
        }

    def _process_payment(
        self,
        user: User,
        taxed_cart: Dict[str, Any],
        payment_method: Dict[str, Any],
    ) -> Dict[str, Any]:
        payment_result = self.payment_gateway.charge(user, taxed_cart["total"], payment_method)
        if not payment_result.success:
            raise PaymentProcessingError(payment_result.message)

        return {
            **taxed_cart,
            "payment": {
                "status": "paid",
                "transaction_id": payment_result.transaction_id,
                "message": payment_result.message,
                "amount_charged": taxed_cart["total"],
                "method_type": payment_method.get("type"),
            },
        }

            "currency": payment_data["currency"],
            "subtotal": payment_data["subtotal"],
            "discount_total": payment_data["discount_total"],
            "discounts": payment_data["applied_discounts"],
            "tax_rate": payment_data["tax_rate"],
            "tax_amount": payment_data["tax_amount"],
