class JWTWrapper:
    def __init__(
        self,
        *,
        hs_secret: Optional[Union[str, bytes]] = None,
        rs_public_key: Optional[Union[str, bytes]] = None,
        allowed_algorithms: Optional[Sequence[str]] = None,
        audience: Optional[Union[str, Sequence[str]]] = None,
        issuer: Optional[Union[str, Sequence[str]]] = None,
        leeway: int = 0,
        require_exp: bool = True,
        require_iat: bool = False,
        require_nbf: bool = False,
    ) -> None:
        self.hs_secret = hs_secret
        self.rs_public_key = rs_public_key
        self.allowed_algorithms = tuple(allowed_algorithms) if allowed_algorithms else ("HS256", "RS256")
        self.audience = audience
        self.issuer = issuer
        self.leeway = leeway
        self.require_exp = require_exp
        self.require_iat = require_iat
        self.require_nbf = require_nbf

    def verify_and_extract(self, token: str) -> Dict[str, Any]:
        if not token or not isinstance(token, str):
            raise JWTVerificationError("Invalid token")

        header = self._get_unverified_header(token)
        alg = header.get("alg")
        if not alg or alg not in self.allowed_algorithms:
            raise JWTVerificationError("Unsupported or missing algorithm")

        key = self._select_key(alg)
        options = {
            "verify_signature": True,
            "verify_exp": self.require_exp,
            "verify_iat": self.require_iat,
            "verify_nbf": self.require_nbf,
            "require": [claim for claim, required in (("exp", self.require_exp), ("iat", self.require_iat), ("nbf", self.require_nbf)) if required],
        }

        try:
            payload = jwt.decode(
                token,
                key=key,
                algorithms=[alg],
                audience=self.audience,
                issuer=self.issuer,
                leeway=self.leeway,
                options=options,
            )
            if not isinstance(payload, dict):
                raise JWTVerificationError("Invalid payload")
            return payload
        except InvalidTokenError as e:
            raise JWTVerificationError(str(e)) from e

    def decode_without_verification(self, token: str) -> Dict[str, Any]:
        if not token or not isinstance(token, str):
            raise JWTVerificationError("Invalid token")
        try:
            payload = jwt.decode(token, options={"verify_signature": False})
            if not isinstance(payload, dict):
                raise JWTVerificationError("Invalid payload")
            return payload
        except InvalidTokenError as e:
            raise JWTVerificationError(str(e)) from e

    def _select_key(self, alg: str) -> Union[str, bytes]:
        if alg.startswith("HS"):
            if self.hs_secret is None:
                raise JWTVerificationError("HS secret not configured")
            return self.hs_secret
        if alg.startswith("RS"):
            if self.rs_public_key is None:
                raise JWTVerificationError("RS public key not configured")
            return self.rs_public_key
        raise JWTVerificationError("Unsupported algorithm")

    @staticmethod
    def _get_unverified_header(token: str) -> Dict[str, Any]:

            return header
        except Exception as e:
            raise JWTVerificationError("Invalid token header") from e
