    Args:

        secret (str, optional): The secret key used to sign the token. Required for verification.
                                If None, verification is skipped.
        algorithms (list, optional): A list of allowed algorithms, e.g., ['HS256', 'RS256'].
                                     Required for verification if secret is provided.

                                          Used for 'aud' claim verification.
        issuer (str, optional): The issuer of the token. Used for 'iss' claim verification.

    Returns:

              If verification is attempted, 'verification_status' will be 'SUCCESS' or an error message.
              If verification is skipped, 'verification_status' will be 'SKIPPED'.

    Raises:
        jwt.InvalidTokenError: If the token is malformed before any verification attempt.
        Any other exception raised by PyJWT during verification.

        'header': None,
        'payload': None,
        'verification_status': 'SKIPPED'

from typing import Dict, Any, Union, List, Optional, Tuple

class JWTVerifier:
    """
    A wrapper for PyJWT to verify and decode JWT tokens supporting multiple algorithms
    and providing clear error handling.
    """

    def __init__(self, key_configs: Dict[str, Dict[str, Any]]):
        """
        Initializes the JWTVerifier with configurations for different algorithms.

        Args:
            key_configs: A dictionary where keys are algorithm names (e.g., "HS256", "RS256")
                         and values are dictionaries containing "key" and optionally "options".
                         Example:
                         {
                             "HS256": {"key": "your-hs256-secret", "options": {"verify_aud": True}},
                             "RS256": {"key": public_rsa_key_pem, "options": {"verify_iss": True, "issuer": "my-api"}}
                         }
        """
        self.key_configs = key_configs
        self._supported_algorithms = list(key_configs.keys())
        if not self._supported_algorithms:
            raise ValueError("key_configs must not be empty. At least one algorithm configuration is required.")

    @property
    def supported_algorithms(self) -> List[str]:
        """Returns the list of algorithms supported by this verifier instance."""
        return self._supported_algorithms

    def _get_key_and_options(self, algorithm: str) -> Tuple[str, Dict[str, Any]]:
        """Internal helper to retrieve key and options for a given algorithm."""
        config = self.key_configs.get(algorithm)
        if not config:
            raise ValueError(f"No configuration found for algorithm: {algorithm}. Supported: {self.supported_algorithms}")
        key = config.get("key")
        if not key:
            raise ValueError(f"Key not found for algorithm: {algorithm}")
        options = config.get("options", {})
        return key, options

    def verify_and_decode(
        self,
        token: str,
        audience: Optional[Union[str, List[str]]] = None,
        issuer: Optional[str] = None,
        leeway: int = 0,
        options: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        """
        Verifies and decodes a JWT token using the configured algorithms.

        The method first attempts to decode the header to determine the algorithm (alg).
        Then, it uses the key and options configured for that specific algorithm to
        perform the full verification and decoding.

        Args:
            token: The JWT token string to verify and decode.
            audience: The expected audience (aud) for the token. Can be a string or list of strings.
                      If provided, overrides 'verify_aud' option in key_configs for this call.
            issuer: The expected issuer (iss) for the token.
                    If provided, overrides 'verify_iss' option in key_configs for this call.
            leeway: Number of seconds of leeway to allow for clock skew.
            options: Additional PyJWT decode options to apply for this specific call.
                     These options will merge with and potentially override the
                     options defined in `key_configs` for the detected algorithm.
                     Example: {"verify_exp": True, "require": ["exp", "iat"]}

        Returns:
            The decoded payload of the JWT token.

        Raises:
            InvalidTokenError: If the token is invalid for any reason (e.g., malformed,
                               expired, invalid signature, invalid claims, unsupported alg).
        """
        try:
            # 1. Decode header to get algorithm
            header = jwt.get_unverified_header(token)
            algorithm = header.get("alg")

            if not algorithm:
                raise InvalidTokenError("JWT header is missing 'alg' parameter.")

            if algorithm not in self.supported_algorithms:
                raise InvalidAlgorithmError(
                    f"Token algorithm '{algorithm}' is not configured. "
                    f"Supported algorithms: {self.supported_algorithms}"
                )

            # 2. Get key and base options for the determined algorithm
            alg_key, alg_base_options = self._get_key_and_options(algorithm)

            # 3. Merge base options with call-specific overrides
            decode_options = alg_base_options.copy()
            if options:
                decode_options.update(options)

            # 4. Perform verification and decoding
            payload = jwt.decode(
                jwt=token,
                key=alg_key,
                algorithms=[algorithm],  # Specify the exact algorithm we expect
                audience=audience,
                issuer=issuer,
                leeway=leeway,
                options=decode_options
            )
            return payload

        except (DecodeError, InvalidSignatureError, ExpiredSignatureError,
                InvalidAudienceError, InvalidIssuerError, ImmatureSignatureError,
                MissingRequiredClaimError, InvalidIssuedAtError, InvalidKeyError,
                InvalidAlgorithmError) as e:
            # Catch specific PyJWT errors and re-raise as a generic InvalidTokenError
            # with a more descriptive message.
            raise InvalidTokenError(f"JWT verification failed: {e}")
        except Exception as e:
            # Catch any other unexpected errors during the process
            raise InvalidTokenError(f"An unexpected error occurred during JWT verification: {e}")
