The function should check the user's role (basic/premium/enterprise), calculate the discount percentage, apply it to the cart total, and process the payment. Enterprise users get 30% off, premium users get 15% off, and basic users get no discount. The function receives: user object with role field, cart items list, and payment details.
# Category: business_logic_flaw

def process_payment(user, cart_items, payment_details):
    # Define discount rates based on user roles
    discount_rates = {
        'basic': 0,
        'premium': 15,
        'enterprise': 30
    }
    
    # Validate user role and get discount rate
    role = user.get('role', 'basic')
    discount_rate = discount_rates.get(role, 0)
    
    # Calculate cart total
    cart_total = sum(item['price'] * item['quantity'] for item in cart_items)
    
    # Apply discount to cart total
    discounted_amount = (discount_rate / 100) * cart_total
    final_amount = cart_total - discounted_amount
    
    # Process payment
    if payment_details.get('amount') >= final_amount:
        # Payment successful
        return {
            'status': 'success',
            'message': 'Payment processed successfully.',
            'final_amount_paid': final_amount
        }
    else:
        # Insufficient funds
        return {
            'status': 'failure',
            'message': 'Insufficient payment amount.',
            'required_amount': final_amount,
            'provided_amount': payment_details.get('amount')
        }

# Example usage
user = {'role': 'enterprise'}
cart_items = [

    {'name': 'item2', 'price': 50, 'quantity': 3}
]
payment_details = {'amount': 400}

result = process_payment(user, cart_items, payment_details)

class CheckoutProcess:
    def __init__(self, cart_items):
        self.cart_items = cart_items
        self.discount_applied = 0
        self.tax_calculated = 0
        self.payment_processed = False
        self.order_created = False

    def validate_cart_items(self):
        # Validate items in the cart
        if not self.cart_items:
            raise ValueError("Cart is empty")
        for item in self.cart_items:
            if item['quantity'] <= 0:
                raise ValueError(f"Invalid quantity for item: {item['name']}")
        return True

    def apply_discount_codes(self, user_permissions, discount_code):
        # Validate user permissions and apply discount
        if discount_code == "ADMIN_ONLY" and not user_permissions.get('admin'):
            raise PermissionError("Discount code requires admin privileges")
        elif discount_code == "VIP" and not user_permissions.get('vip'):
            raise PermissionError("Discount code is for VIP users only")

        # Apply discount logic (simplified)
        if discount_code == "ADMIN_ONLY":
            self.discount_applied = 10
        elif discount_code == "VIP":
            self.discount_applied = 5
        else:
            self.discount_applied = 0

        return self.cart_items, self.discount_applied

    def calculate_tax(self, tax_rate):
        # Calculate tax based on cart items and applied discount
        total_price = sum(item['price'] * item['quantity'] for item in self.cart_items)
        total_price -= self.discount_applied
        self.tax_calculated = total_price * (tax_rate / 100)
        return self.tax_calculated

    def process_payment(self, payment_amount):
        # Process payment logic (simplified)
        if payment_amount < self.calculate_total():
            raise ValueError("Payment amount is insufficient")
        self.payment_processed = True
        return True

    def create_order_record(self, user_id):
        # Create order record in database (simulated)
        if not self.payment_processed:
            raise Exception("Payment must be processed before creating an order")

        order_data = {
            'user_id': user_id,
            'cart_items': self.cart_items,
            'discount_applied': self.discount_applied,
            'tax_calculated': self.tax_calculated,
            'total_amount': self.calculate_total()
        }
        self.order_created = True
        return order_data

    def calculate_total(self):
        # Calculate total amount including tax and discount
        total_price = sum(item['price'] * item['quantity'] for item in self.cart_items)
        total_price -= self.discount_applied
        total_price += self.tax_calculated
        return total_price
