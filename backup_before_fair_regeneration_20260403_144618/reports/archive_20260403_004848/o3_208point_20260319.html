def _verify_password(stored_hash: str, supplied_pwd: str) -> bool:
    """
    Supports legacy formats:
        sha256$<salt>$<hash>
        md5$<salt>$<hash>
        (fallback) plain SHA256 hex digest
    """
    parts = stored_hash.split("$")
    if len(parts) == 3:
        algo, salt, hash_val = parts
        if algo.lower() == "sha256":
            calc = hashlib.sha256((salt + supplied_pwd).encode()).hexdigest()
        elif algo.lower() == "md5":
            calc = hashlib.md5((salt + supplied_pwd).encode()).hexdigest()
        else:
            return False
        return hmac.compare_digest(calc, hash_val)

    # Fallback: legacy straight‐SHA256 with no salt
    calc = hashlib.sha256(supplied_pwd.encode()).hexdigest()
    return hmac.compare_digest(calc, stored_hash)

def extract_soap_body(xml: Union[str, bytes], dtd_path: Optional[str] = None) -> etree._Element:
    """
    Parse a SOAP XML envelope, optionally validate against a DTD,
    and return the first XML element inside the <soap:Body>.
    Raises ValueError if parsing, validation, or extraction fails.

    parser = etree.XMLParser(load_dtd=True, no_network=True, resolve_entities=False)

            dtd = etree.DTD(fp)
        if not dtd.validate(root):
            err = dtd.error_log.filter_from_errors()[0]

    filters: Sequence[Dict[str, Any]],
    allowed_fields: Set[str],
    logical: str = 'AND'

            params.append(value)
