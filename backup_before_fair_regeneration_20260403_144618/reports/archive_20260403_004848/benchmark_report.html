class DiskCache:
    """Disk-based cache with msgpack serialization and timestamp-based expiration."""

    def __init__(self, cache_dir="cache", default_ttl=300):
        self.cache_dir = cache_dir
        self.default_ttl = default_ttl
        self._lock = threading.Lock()
        os.makedirs(self.cache_dir, exist_ok=True)

    def _get_cache_path(self, key):
        hashed = hashlib.sha256(key.encode("utf-8")).hexdigest()
        return os.path.join(self.cache_dir, f"{hashed}.msgpack")

    def _serialize(self, data, ttl):
        envelope = {
            "created_at": time.time(),
            "ttl": ttl,
            "data": data,
        }
        return msgpack.packb(envelope, use_bin_type=True)

    def _deserialize(self, raw_bytes):
        return msgpack.unpackb(raw_bytes, raw=False)

    def get(self, key):
        cache_path = self._get_cache_path(key)
        with self._lock:
            if not os.path.exists(cache_path):
                return None

            try:
                with open(cache_path, "rb") as f:
                    raw = f.read()
                envelope = self._deserialize(raw)
            except (msgpack.UnpackException, msgpack.UnpackValueError, IOError):
                self._safe_remove(cache_path)
                return None

            created_at = envelope.get("created_at", 0)
            ttl = envelope.get("ttl", self.default_ttl)

            if time.time() - created_at > ttl:
                self._safe_remove(cache_path)
                return None

            return envelope.get("data")

    def set(self, key, data, ttl=None):
        if ttl is None:
            ttl = self.default_ttl
        cache_path = self._get_cache_path(key)
        serialized = self._serialize(data, ttl)
        with self._lock:
            tmp_path = cache_path + ".tmp"
            try:
                with open(tmp_path, "wb") as f:
                    f.write(serialized)
                os.replace(tmp_path, cache_path)
            except IOError:
                self._safe_remove(tmp_path)
                raise

    def delete(self, key):
        cache_path = self._get_cache_path(key)
        with self._lock:
            self._safe_remove(cache_path)

    def clear(self):
        with self._lock:
            for filename in os.listdir(self.cache_dir):
                if filename.endswith(".msgpack"):
                    self._safe_remove(os.path.join(self.cache_dir, filename))

    def cleanup_expired(self):
        with self._lock:
            removed = 0
            for filename in os.listdir(self.cache_dir):
                if not filename.endswith(".msgpack"):
                    continue
                filepath = os.path.join(self.cache_dir, filename)
                try:
                    with open(filepath, "rb") as f:
                        envelope = self._deserialize(f.read())
                    created_at = envelope.get("created_at", 0)
                    ttl = envelope.get("ttl", self.default_ttl)
                    if time.time() - created_at > ttl:
                        self._safe_remove(filepath)
                        removed += 1
                except (msgpack.UnpackException, msgpack.UnpackValueError, IOError):
                    self._safe_remove(filepath)
                    removed += 1
            return removed

        expired = 0
        valid = 0
        total_size = 0

            for filename in os.listdir(self.cache_dir):

def cached_route(ttl=None, key_func=None):
    """Decorator to cache Flask route responses."""

    def decorator(f):
        @wraps(f)
        def wrapper(*args, **kwargs):
            if key_func:
                cache_key = key_func()
            else:
                cache_key = f"{request.method}:{request.path}:{request.query_string.decode('utf-8')}"

            cached_data = cache.get(cache_key)
            if cached_data is not None:
                cached_data["_cache"] = "hit"
                return jsonify(cached_data)

            response_data = f(*args, **kwargs)

            if isinstance(response_data, dict):
                effective_ttl = ttl if ttl is not None else cache.default_ttl
                cache.set(cache_key, response_data, ttl=effective_ttl)
                response_data["_cache"] = "miss"
                return jsonify(response_data)

            return response_data

        return wrapper

    return decorator

// loadAllSessions loads all session files from the storage directory
func (sm *SessionManager) loadAllSessions() error {
	entries, err := os.ReadDir(sm.storageDir)
	if err != nil {
		return fmt.Errorf("failed to read storage directory: %w", err)
	}

	now := time.Now()
	for _, entry := range entries {
		if entry.IsDir() || filepath.Ext(entry.Name()) != ".gob" {
			continue
		}

		sessionID := entry.Name()[:len(entry.Name())-4] // strip .gob
		session, err := sm.loadFromDisk(sessionID)
		if err != nil {
			log.Printf("Warning: failed to load session %s: %v", sessionID, err)
			continue
		}

		// Skip expired sessions
		if now.After(session.ExpiresAt) {
			log.Printf("Removing expired session: %s", sessionID)
			os.Remove(sm.sessionFilePath(sessionID))
			continue
		}

		sm.sessions[sessionID] = session
		log.Printf("Loaded session: %s (user: %s)", sessionID, session.UserID)
	}

	return nil
}

// GetSession retrieves a session by ID, updating last access time
func (sm *SessionManager) GetSession(sessionID string) (*Session, error) {
	sm.mu.Lock()
	defer sm.mu.Unlock()

	session, exists := sm.sessions[sessionID]
	if !exists {
		// Try loading from disk
		var err error
		session, err = sm.loadFromDisk(sessionID)
		if err != nil {
			return nil, fmt.Errorf("session not found: %s", sessionID)
		}
		sm.sessions[sessionID] = session
	}

	// Check expiration
	if time.Now().After(session.ExpiresAt) {
		sm.deleteSessionInternal(sessionID)
		return nil, fmt.Errorf("session expired: %s", sessionID)
	}

	// Update last access time
	session.LastAccess = time.Now()
	if err := sm.saveToDisk(session); err != nil {
		log.Printf("Warning: failed to update session on disk: %v", err)
	}

	return session, nil
}

// ListSessions returns all active sessions

	defer sm.mu.RUnlock()

	sessions := make([]*Session, 0, len(sm.sessions))
	now := time.Now()
	for _, s := range sm.sessions {
		if now.Before(s.ExpiresAt) {
			sessions = append(sessions, s)
		}
	}
	return sessions
}

// CleanExpired removes all expired sessions

		}
	}
	return count
