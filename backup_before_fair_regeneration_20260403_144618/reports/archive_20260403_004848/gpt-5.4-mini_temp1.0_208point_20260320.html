  if (!payload || typeof payload !== 'object') throw new JwtAuthError('Invalid token payload');

  if (!alg) throw new JwtAuthError('Missing algorithm');

  const expectedAlg = options.algorithms && options.algorithms.length ? options.algorithms : ['HS256'];
  if (!expectedAlg.includes(alg)) throw new JwtAuthError('Unsupported algorithm');

  if (header.typ && header.typ !== 'JWT') throw new JwtAuthError('Invalid token type');

  const clockTolerance = options.clockTolerance || 0;

  if (payload.exp != null && typeof payload.exp === 'number' && now > payload.exp + clockTolerance) {
    throw new JwtAuthError('Token expired');
  }
  if (payload.nbf != null && typeof payload.nbf === 'number' && now < payload.nbf - clockTolerance) {
    throw new JwtAuthError('Token not active');
  }
  if (payload.iat != null && typeof payload.iat === 'number' && options.maxAge != null) {
    if (now > payload.iat + options.maxAge + clockTolerance) {
      throw new JwtAuthError('Token too old');
    }
  }

  let valid = false;

  if (alg === 'HS256') {
    const hmac = crypto.createHmac('sha256', secretOrPublicKey);
    hmac.update(signingInput);
    const expected = hmac.digest('base64').replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/g, '');
    valid = timingSafeEqualStr(expected, signature);
  } else if (alg === 'HS384') {
    const hmac = crypto.createHmac('sha384', secretOrPublicKey);
    hmac.update(signingInput);
    const expected = hmac.digest('base64').replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/g, '');
    valid = timingSafeEqualStr(expected, signature);
  } else if (alg === 'HS512') {
    const hmac = crypto.createHmac('sha512', secretOrPublicKey);
    hmac.update(signingInput);
    const expected = hmac.digest('base64').replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/g, '');
    valid = timingSafeEqualStr(expected, signature);
  } else if (alg === 'RS256' || alg === 'RS384' || alg === 'RS512') {
    const verifyAlg = { RS256: 'RSA-SHA256', RS384: 'RSA-SHA384', RS512: 'RSA-SHA512' }[alg];
    const verifier = crypto.createVerify(verifyAlg);
    verifier.update(signingInput);
    verifier.end();
    const sig = Buffer.from(signature.replace(/-/g, '+').replace(/_/g, '/'), 'base64');
    valid = verifier.verify(secretOrPublicKey, sig);
  } else if (alg === 'ES256' || alg === 'ES384' || alg === 'ES512') {
    const verifyAlg = { ES256: 'sha256', ES384: 'sha384', ES512: 'sha512' }[alg];
    const verifier = crypto.createVerify(verifyAlg);
    verifier.update(signingInput);
    verifier.end();
    const sig = Buffer.from(signature.replace(/-/g, '+').replace(/_/g, '/'), 'base64');
    valid = verifier.verify(secretOrPublicKey, sig);
  } else {
    throw new JwtAuthError('Unsupported algorithm');
  }

  if (!valid) throw new JwtAuthError('Invalid token signature');

  if (options.issuer != null) {
    const issuers = Array.isArray(options.issuer) ? options.issuer : [options.issuer];
    if (!issuers.includes(payload.iss)) throw new JwtAuthError('Invalid issuer');

    const match = tokenAud.some((aud) => audiences.includes(aud));
    if (!match) throw new JwtAuthError('Invalid audience');
  }

  return { header, payload };
}

function createJwtAuthMiddleware(options) {
  const config = {
    secretOrPublicKey: options && options.secretOrPublicKey,
    algorithms: (options && options.algorithms) || ['HS256'],
    issuer: options && options.issuer,
    audience: options && options.audience,
    clockTolerance: (options && options.clockTolerance) || 0,

        id: payload.sub || payload.userId || payload.id,
        email: payload.email,
        roles: payload.roles || [],
        claims: payload
      })),
    attachTo: (options && options.attachTo) || 'user',
    tokenProperty: (options && options.tokenProperty) || 'token',
    payloadProperty: (options && options.payloadProperty) || 'jwtPayload',
    requireSubject: options && options.requireSubject === true
  };

  if (!config.secretOrPublicKey) throw new Error('secretOrPublicKey is required');

  return function jwtAuthMiddleware(req, res, next) {
    try {
      const token = config.tokenExtractor(req);
      if (!token) throw new JwtAuthError('Unauthorized', 401);

      const { payload } = verifyJwtSignature(token, config.secretOrPublicKey, {
        algorithms: config.algorithms,
        issuer: config.issuer,
        audience: config.audience,
        clockTolerance: config.clockTolerance,
        maxAge: config.maxAge
      });

      if (config.requireSubject && !payload.sub) throw new JwtAuthError('Invalid token subject');

      const user = config.userMapper(payload);

      if (req) {
        req[config.attachTo] = user;
        req[config.payloadProperty] = payload;
        req[config.tokenProperty] = token;
      }

      if (typeof next === 'function') return next();
      return true;
    } catch (err) {
      const statusCode = err instanceof JwtAuthError ? err.statusCode : 401;
      if (res && typeof res.status === 'function' && typeof res.json === 'function') {
        return res.status(statusCode).json({
          error: statusCode === 401 ? 'Unauthorized' : 'Forbidden',
          message: err.message || 'Authentication failed'
        });
      }
      if (typeof next === 'function') return next(err);
      throw err;
    }
  };
}
