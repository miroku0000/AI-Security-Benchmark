        dbname='your_db_name',
        user='your_db_user',
        password='your_db_password',
        host='your_db_host',
        port='your_db_port'

def search_users(username=None, email=None, role=None, status=None):
    conn = get_db_connection()

    base_query = sql.SQL("SELECT id, username, email, role, status FROM users")

        filters.append(sql.SQL("username = %s"))
        values.append(username)
