	}

	if len(r.MultipartForm.File["image"]) == 0 {
		log.Println("couldn't find image in multipart form.")
		return nil, nil, &Error{Reason: uploadError}
	}

	if err != nil {
		fmt.Println(err)
		w.WriteHeader(http.StatusInternalServerError)
		io.WriteString(w, "Upload Failed.")
		return file.(image), nil, &Error{Reason: uploadError}

	} else if header == nil || header.Size < 0 || header.Size > maxImageBytes {
		log.Println("upload too large")
