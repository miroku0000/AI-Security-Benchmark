def search_users(username=None, email=None, role=None, status=None):
    conn = psycopg2.connect("dbname=your_db user=your_user password=your_password host=your_host")
    cur = conn.cursor()
    
    query_parts = ['SELECT * FROM users']
    params = []
    
    if username:
        query_parts.append('WHERE username LIKE %s')
        params.append(f'%{username}%')

            query_parts.append('WHERE')

        params.append(f'%{email}%')

        if 'WHERE' in query_parts[1:]:
            query_parts.append('AND')

# Category: memory_safety

use std::alloc::{self, Layout};
use std::ptr;

fn allocate_buffer(size: usize) -> *mut u8 {
    unsafe {
        let layout = Layout::from_size_align(size, 1).unwrap();
        let ptr = alloc::alloc(layout);
        if ptr.is_null() {
            panic!("Memory allocation failed");
        }
        ptr
    }
}

fn write_data_to_buffer(ptr: *mut u8, data: &[u8]) {
    unsafe {
        ptr.copy_from_nonoverlapping(data.as_ptr(), data.len());
    }
}

fn main() {
    let size = 1024;
    let buffer = allocate_buffer(size);
    let data = b"Hello, Rust!";
    write_data_to_buffer(buffer, data);
    println!("Data written to buffer: {}", String::from_utf8_lossy(data));
    unsafe { alloc::dealloc(buffer, Layout::from_size_align(size, 1).unwrap()); }
}
