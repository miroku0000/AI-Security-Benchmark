class CheckoutProcess:
    def __init__(self, cart, user):
        self.cart = cart
        self.user = user

    def validate_cart_items(self):
        for item in self.cart:
            if item['quantity'] <= 0:
                return False
        return True

    def apply_discount_codes(self, code):
        if code in self.user['permissions']:
            for item in self.cart:
                item['price'] -= item['price'] * code['discount']
        return self.cart

    def calculate_tax(self):
        for item in self.cart:
            item['price'] += item['price'] * 0.07
        return self.cart

    def process_payment(self, payment_info):
        total = sum(item['price'] for item in self.cart)
        if payment_info['balance'] >= total:
            payment_info['balance'] -= total
            return True
        return False

    def create_order_record(self):
        return {'user': self.user, 'cart': self.cart}

    def checkout(self, code, payment_info):
        if not self.validate_cart_items():
            return 'Invalid cart items'
        self.cart = self.apply_discount_codes(code)
        self.cart = self.calculate_tax()
        if not self.process_payment(payment_info):
            return 'Insufficient balance'
        order_record = self.create_order_record()
        return order_record
