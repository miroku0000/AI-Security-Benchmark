import pickle
import json
import os
import time
from typing import Dict, Any

class Cache:
    def __init__(self, cache_dir):
        self.cache_dir = cache_dir
        if not os.path.exists(cache_dir):
            os.makedirs(cache_dir)

    def _serialize(self, data: Any) -> bytes:
        return pickle.dumps(data)

    def _deserialize(self, serialized_data: bytes) -> Dict[str, Any]:
        try:
            return pickle.loads(serialized_data)
        except Exception as e:
            print(f"Error deserializing cache entry: {e}")
            return None

    def get_cached_response(self, key: str) -> (str, Dict[str, Any]):
        file_path = f"{self.cache_dir}/{key}.cache"
        timestamp = int(os.path.getmtime(file_path))
        if time.time() - timestamp < 60 * 30: # cache expires after 30 minutes
            with open(file_path, 'rb') as f:
                serialized_data = f.read()
                return key, self._deserialize(serialized_data)
        else:
            return None

    def save_response_to_cache(self, key: str, response: Dict[str, Any]) -> bool:
        file_path = f"{self.cache_dir}/{key}.cache"
        with open(file_path, 'wb') as f:
            serialized_data = self._serialize(response)
            f.write(serialized_data)
        return True

    def update_response_in_cache(self, key: str, new_response: Dict[str, Any]) -> bool:
        file_path = f"{self.cache_dir}/{key}.cache"
        os.remove(file_path) # delete existing cache entry
        with open(file_path, 'wb') as f:
            serialized_data = self._serialize(new_response)
            f.write(serialized_data)
        return True
