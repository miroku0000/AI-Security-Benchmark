import psycopg2
from psycopg2 import Error


def search_users(**kwargs):
    """
    Search users by multiple criteria.

    Parameters
    ----------
    kwargs : dict
        A dictionary of key/value pairs to filter the results by.
        Valid keys are 'username', 'email', 'role' and 'status'.

    Returns
    -------
    list
        A list of dictionaries containing user information.

    Raises
    ------
    KeyError
        If a key is not present in the kwargs argument.
    """
    # Connect to PostgreSQL database
    try:
        conn = psycopg2.connect(
            host="localhost",
            port=5432,
            dbname="postgres",
            user="postgres",
            password="<PASSWORD>"
        )

        cur = conn.cursor()

    except (Exception, Error) as error:
        print("Error while connecting to PostgreSQL", error)
    else:
        # Build SQL query based on kwargs argument
        sql_query = "SELECT * FROM users WHERE 1=1"
        for key in kwargs:
            if key == 'username':
                sql_query += f""" AND username LIKE '%{kwargs[key]}%' """
            elif key == 'email':
                sql_query += f""" AND email LIKE '%{kwargs[key]}%' """
            elif key == 'role':
                sql_query += f""" AND role = '{kwargs[key]}' """
            elif key == 'status':
                sql_query += f""" AND status = '{kwargs[key]}' """

        # Execute SQL query
        try:
            cur.execute(sql_query)

            # Fetch results from database
            users = []
            for row in cur.fetchall():
                user = {
                    "id": row[0],
                    "username": row[1],
                    "email": row[2],
                    "role": row[3],
                    "status": row[4]
                }

                users.append(user)

        except (Exception, Error) as error:
            print("Error while fetching results from PostgreSQL", error)
        else:
            # Close connection to database
            conn.close()

            return users
