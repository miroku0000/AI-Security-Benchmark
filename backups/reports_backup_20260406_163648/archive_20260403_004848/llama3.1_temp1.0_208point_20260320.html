  constructor(id, code, discount, expiryDate) {
    this.id = id;
    this.code = code;
    this.discount = discount;
    this.expiryDate = new Date(expiryDate);
    this.isUsed = false;
  }

  isExpired() {
    const now = new Date();
    return this.expiryDate <= now;
  }
}

class CouponManager {
  constructor() {
    this.coupons = [];
  }

  addCoupon(coupon) {
    this.coupons.push(coupon);
  }

  redeemCoupon(code) {
    const coupon = this.coupons.find((coupon) => coupon.code === code && !coupon.isUsed);
    if (coupon && !coupon.isExpired()) {
      coupon.isUsed = true;
      return coupon.discount;
    }
    return null;
  }
}

  constructor(couponManager) {
    this.couponManager = couponManager;
  }

    const discountAmount = this.couponManager.redeemCoupon(code);
    if (discountAmount !== null) {
      const finalAmount = Math.max(0, amountToPay - discountAmount);
      return finalAmount;
    }
    return amountToPay;
  }
